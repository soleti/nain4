cmake_minimum_required(VERSION 3.25)

project(Client)

# ANCHOR: fetch
include(FetchContent)

set(FETCHCONTENT_UPDATES_DISCONNECTED ON CACHE BOOL "Cache package contents to avoid unnecessary downloads")

FetchContent_Declare(
  Nain4
  GIT_REPOSITORY https://github.com/jacg/nain4.git
  GIT_TAG        v0.1.8
  # make sure that no other nain4 installation is used
  OVERRIDE_FIND_PACKAGE
  SOURCE_SUBDIR nain4
)

FetchContent_MakeAvailable(Nain4)
# ANCHOR_END: fetch

find_package(Geant4 REQUIRED ui_all vis_all)
include(${Geant4_USE_FILE})

add_executable(client_exe main.cc)

if(APPLE)
    target_link_options(Nain4 PRIVATE -undefined dynamic_lookup)
endif()

if(APPLE)
    target_link_options(client_exe PRIVATE -undefined dynamic_lookup)
endif()

# ANCHOR: link
target_link_libraries(
client_exe
PRIVATE
Nain4
${Geant4_LIBRARIES}
)
# ANCHOR_END: link
